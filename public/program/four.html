#include<stdio.h>
#include<stdlib.h>


int a,b,i,j,n,ne=1;
int visited[10] = {0};
int min,mincost=0,cost[10][10],ind[10];

int main(){
	printf("\nEnter the number of vertices : ");
	scanf("%d",&n);
	if(n<=0){
		printf("Number of vertices cant be zero or less than zero \n");
		exit(0);
	}

	printf("\nEnter the value in adjacency matrix : \n");
	for(i=1;i<=n;i++){
		printf("\nRow %d elements : ",i);
		for(j=1;j<=n;j++){
			scanf("%d",&cost[i][j]);
			if(cost[i][j]==0)
				cost[i][j]=999;
		}
	}
	for(i=1;i<=n;i++){
		ind[i]=0;
	}
	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++){
			if(cost[i][j]==999){
				
				ind[i]=ind[i]+0;
			}
			else{
				ind[i]=ind[i]+cost[j][i];
			}
			
		}
	}
	for(i=1;i<=n;i++){
		if(ind[i]==0){
			printf("%c has no connections \n",i+64);
			exit(0);
		}
	}
	visited[1]=1;

	while(ne<n){
		for(i=1,min=999;i<=n;i++)
			for(j=1;j<=n;j++)
				if(cost[i][j]<min)
					if(visited[i]!=0){
						min = cost[i][j];
						a = i;
						b = j;
					}
		if(visited[a] == 0 || visited[b] == 0){
			printf("\n %d: (%d %d) is cost : %d ",ne++,a,b,min);
			mincost += min;
			visited[b] = 1;
		}

		cost[a][b] = cost[b][a] = 999;
	}
	printf("\nMin cost : %d  \n",mincost);
	return(0);
}
